class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int left=0,right=nums.size()-1;
        int mid=(left+right)>>1;
        while(left<=right)
        {
            if (nums[mid]==target)
               return mid;
            else if (nums[mid]<target)
               left=mid+1;
            else
               right=mid-1;   
            mid=(left+right)>>1;    
        }
        return left;//这里要想明白为什么是left，其实还是按照左闭右闭来思考，这里记住也可以

    }
};
